219 double CalcCircle(GUPoint<short> *pt1, GUPoint<short> *pt2, GUPoint<short> *pt3) {
220 double yDelta_a= pt2->y - pt1->y;
221 double xDelta_a= pt2->x - pt1->x;
222 double yDelta_b= pt3->y - pt2->y;
223 double xDelta_b= pt3->x - pt2->x;
224
225 if (fabs(xDelta_a) <= 0.000000001 && fabs(yDelta_b) <= 0.000000001){
226 this->m_Center.x= short(0.5*(pt2->x + pt3->x));
227 this->m_Center.y= short(0.5*(pt1->y + pt2->y));
228 this->m_dRadius= GUPoint<short>::Length(this->m_Center, *pt1);// calc. radius
229
230 return this->m_dRadius;
231 }
232
233 // IsPerpendicular() assure that xDelta(s) are not zero
234 double aSlope=yDelta_a/xDelta_a; //
235 double bSlope=yDelta_b/xDelta_b;
236 if (fabs(aSlope-bSlope) <= 0.000000001){ // checking whether the given points are colinear.
237 this->m_dRadius = -1;
238 return -1;
239 }
240
241 double cx, cy, r;
242 cx = (aSlope*bSlope*(pt1->y - pt3->y) + bSlope*(pt1->x + pt2 ->x)
243 - aSlope*(pt2->x+pt3->x) )/(2.0* (bSlope-aSlope) );
244 cy = -1.0*(cx - (pt1->x+pt2->x)/2.0)/aSlope + (pt1->y+pt2->y)/2.0;
245 r = GUPoint<double>::Length(GUPoint<double>(cx, cy), GUPoint<double>(static_cast<double>(pt1->x), static_cast<double>(pt1->y)));
246 // calc center
247 this->m_Center.x= short(cx + 0.5);
248 this->m_Center.y = short(cy + 0.5);
249
250 this->m_dRadius= r; // calc. radius
251
252 //fprintf(stderr, "cx %f cy %f r %f\n", cx, cy, r);
253
254 return this->m_dRadius;
