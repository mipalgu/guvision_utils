33void ROTATE_POINT(double &pointX, double &pointY, double rotateAngle, double rotateAboutPointX /*= 0*/, double rotateAboutPointY /*= 0*/)
34{
35 double cosTheta = cos(rotateAngle);
36 double sinTheta = sin(rotateAngle);
37
38 TRANSLATE_POINT(pointX, pointY, rotateAboutPointX, rotateAboutPointY);
39 double newX = cosTheta * pointX - sinTheta * pointY;
40 double newY = sinTheta * pointX + cosTheta * pointY;
41 pointX = newX;
42 pointY = newY;
43 TRANSLATE_POINT(pointX, pointY, -1 * rotateAboutPointX, -1 * rotateAboutPointY);
44}

46void ROTATE_IMAGE_POINT(double &pointX, double &pointY, double rotateAngle, int imgY)
47{
48 double oldAngle = atan((imgY - pointY) / pointX);
49 if (pointX < 0) oldAngle += PI;
50 double newAngle = oldAngle + rotateAngle;
51 double r = SQRT(SQUARE(pointX) + SQUARE(imgY - pointY));
52
53 pointX = r * cos(newAngle);
54 pointY = imgY - (r * sin(newAngle));
55}

65double SQ_POLAR_DISTANCE(double r1, double theta1, double r2, double theta2)
66{
67 double ang = MAX(theta1, theta2) - MIN(theta1, theta2);
68 double cang = cos(ang);
69 double dist = SQUARE(r1) + SQUARE(r2) + (2 * r1 * r2 * cang);
70 return dist;
71}
